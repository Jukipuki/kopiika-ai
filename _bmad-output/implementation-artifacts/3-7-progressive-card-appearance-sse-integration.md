# Story 3.7: Progressive Card Appearance & SSE Integration

Status: done

## Story

As a **user**,
I want to see insight cards appear progressively as the AI pipeline processes my data,
So that I experience an engaging, real-time reveal rather than a static loading screen.

## Acceptance Criteria

1. **Given** I have just uploaded a statement and the pipeline is processing, **When** the AI agents complete their work, **Then** cards fade/slide in one by one as each insight is generated (basic categorization insights appear first, then deeper education cards).

2. **Given** the SSE stream is active, **When** new insights are generated by the pipeline, **Then** the frontend receives `insight-ready` events and appends new cards to the `CardStackNavigator` with animation.

3. **Given** progressive loading is happening, **When** the UI displays the Teaching Feed, **Then** a subtle animation and "AI is still thinking..." indicator shows while insights are being generated.

4. **Given** loading messages are displayed, **When** the user reads them, **Then** they see lighthearted copy such as "Crunching your numbers...", "Finding patterns in your spending...", "Almost there...".

### SSE Architecture (already built in Story 2.6)

The SSE stream at `GET /api/v1/jobs/{id}/stream?token=<JWT>` emits:
```
event: pipeline-progress
data: {"event": "pipeline-progress", "jobId": "...", "step": "parsing|categorization|education", "progress": 20|60|80, "message": "..."}

event: insight-ready          ← NEW (Story 3.7 adds this)
data: {"event": "insight-ready", "jobId": "...", "insightId": "...", "type": "<category>"}

event: job-complete
data: {"event": "job-complete", "jobId": "...", "status": "completed", "totalInsights": 12, ...}
```

Auth: EventSource doesn't support Authorization headers. Use `?token=<JWT>` query param — same pattern established in `use-job-status.ts`.

**File:** `backend/app/tasks/processing_tasks.py` (lines ~186–202)

**Current pattern (batch commit, no per-insight events):**

**New pattern (collect refs, emit per-insight after commit):**
```python
insight_cards = result_state.get("insight_cards", [])
insights: list[Insight] = []
if insight_cards:
    for card in insight_cards:
        insight = Insight(
            user_id=job.user_id,
            upload_id=upload.id,
            headline=card.get("headline", ""),
            key_metric=card.get("key_metric", ""),
            why_it_matters=card.get("why_it_matters", ""),
            deep_dive=card.get("deep_dive", ""),
            severity=card.get("severity", "medium"),
            category=card.get("category", "other"),
        )
        session.add(insight)
        insights.append(insight)
    session.commit()  # IDs assigned after this point
    for insight in insights:
        publish_job_progress(job_id, {
            "event": "insight-ready",
            "jobId": job_id,
            "insightId": str(insight.id),
            "type": insight.category,
        })
```

**Note:** SQLModel/SQLAlchemy populates `insight.id` after `session.commit()` (or `session.refresh(insight)` if using lazy loading). The current model uses `uuid.uuid4()` as default_factory on the `id` field, so IDs are actually set at object creation time — double-check with `session.refresh` if needed.

### Frontend — `use-feed-sse` Hook Pattern

```typescript
// frontend/src/features/teaching-feed/hooks/use-feed-sse.ts
"use client";

import { useState, useEffect, useRef, useCallback } from "react";
import { useQueryClient } from "@tanstack/react-query";

const API_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

export function useFeedSSE(jobId: string | null, accessToken: string | undefined) {
  const queryClient = useQueryClient();
  const [pendingInsightIds, setPendingInsightIds] = useState<string[]>([]);
  const [isStreaming, setIsStreaming] = useState(false);
  const [phase, setPhase] = useState<string | null>(null);
  const esRef = useRef<EventSource | null>(null);
  const terminalRef = useRef(false);

  const cleanup = useCallback(() => {
    esRef.current?.close();
    esRef.current = null;
  }, []);

  useEffect(() => {
    if (!jobId || !accessToken) {
      setIsStreaming(false);
      return;
    }

    terminalRef.current = false;
    setIsStreaming(true);
    setPendingInsightIds([]);
    setPhase(null);

    const url = `${API_URL}/api/v1/jobs/${jobId}/stream?token=${encodeURIComponent(accessToken)}`;
    const es = new EventSource(url);
    esRef.current = es;

    es.addEventListener("pipeline-progress", (e: MessageEvent) => {
      const data = JSON.parse(e.data);
      setPhase(data.step ?? null);
    });

    es.addEventListener("insight-ready", (e: MessageEvent) => {
      const data = JSON.parse(e.data);
      if (data.insightId) {
        setPendingInsightIds((prev) => [...prev, data.insightId]);
      }
    });

    es.addEventListener("job-complete", () => {
      terminalRef.current = true;
      setIsStreaming(false);
      setPhase(null);
      queryClient.invalidateQueries({ queryKey: ["teaching-feed"] });
      cleanup();
    });

    es.addEventListener("job-failed", () => {
      terminalRef.current = true;
      setIsStreaming(false);
      setPendingInsightIds([]);
      cleanup();
    });

    es.onerror = () => {
      if (!terminalRef.current) {
        setIsStreaming(false);
      }
      cleanup();
    };

    return cleanup;
  }, [jobId, accessToken, cleanup, queryClient]);

  return { pendingInsightIds, isStreaming, phase };
}
```

```typescript
// In test file or vitest.setup.ts (check frontend/vitest.setup.ts or frontend/src/test/setup.ts)
class MockEventSource {
  static instances: MockEventSource[] = [];
  url: string;
  onopen: (() => void) | null = null;
  onerror: ((e: Event) => void) | null = null;
  private listeners: Record<string, ((e: MessageEvent) => void)[]> = {};

  constructor(url: string) {
    this.url = url;
    MockEventSource.instances.push(this);
  }

  addEventListener(type: string, handler: (e: MessageEvent) => void) {
    this.listeners[type] = [...(this.listeners[type] ?? []), handler];
  }

  emit(type: string, data: unknown) {
    const event = { data: JSON.stringify(data) } as MessageEvent;
    this.listeners[type]?.forEach((h) => h(event));
  }

  close() {}
}
```
